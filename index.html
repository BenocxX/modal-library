Description

                            <button class="myButton" data-modal-launcher
                                        data-modal-id="myModal">
                                Button
                            </button>
                            
                            <div class="modal" id="exampleModal1">
                                <div class="modal-dialog">
                                    <div class="modal-header-container">
                                        <h3 class="modal-title">Title</h3>
                                        <button data-close-button class="modal-button">X</button>
                                    </div>
                                    <div class="modal-content-container">
                                        <p class="modal-text">Text</p>
                                    </div>
                                    <div class="modal-footer-container">
                                        <button data-close-button class="modal-button">Close</button>
                                        <button class="modal-button">Test</button>
                                    </div>
                                </div>
                            </div>
                            
                            <div class="modal-shadow"></div>

                            const myButton = document.querySelector("[data-modal-launcher]")

Customizing elements in the modal

Description 


                            // JavaScript
                            const modal = document.querySelector("#myModal")
                            modal.addEventListener("modal.shown", function (e) {
                                const dialog = this.querySelector(".modal-dialog")
                                const dialogTitle = dialog.querySelector("h3")
                                const dialogText = dialog.querySelector("p")
                            
                                dialogTitle.innerText = "My custom title"
                                dialogText.innerText = "My custom text"
                            })

Passing info through modal

Description 


                            // JavaScript
                            const myButton = document.querySelector("#myButton")
                                .addEventListener("click", (e) => {
                                    onModalLaunch(e, {
                                        title: "Where am I from?",
                                        text: "I am text coming from the \"detail\" parameter!"
                                    })
                                })
                            
                            const modal = document.querySelector("#myModal")
                            modal.addEventListener("modal.shown", function (e) {
                                const dialog = this.querySelector(".modal-dialog")
                                const dialogTitle = dialog.querySelector("h3")
                                const dialogText = dialog.querySelector("p")
                            
                                dialogTitle.innerText = e.detail.title
                                dialogText.innerText = e.detail.text
                            })
